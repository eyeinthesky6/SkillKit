## üîß **Technical Architecture**

### **Universal Workflow Engine**
```typescript
export class UniversalWorkflowEngine {
  private adapters: Map<string, IDEAdapter> = new Map();
  private assistants: Map<string, AIAssistant> = new Map();

  async executeWorkflow(workflowName: string, context: EditorContext) {
    // Detect current IDE
    const ide = await this.detectIDE();

    // Get appropriate adapter
    const adapter = this.adapters.get(ide);
    if (!adapter) {
      throw new Error(`IDE ${ide} not supported`);
    }

    // Execute workflow using IDE-specific adapter
    await adapter.executeWorkflow(workflowName, context);
  }

  async detectIDE(): Promise<string> {
    // Check environment variables, process info, etc.
    if (process.env.CURSOR) return 'cursor';
    if (process.env.VSCODE_PID) return 'vscode';
    // ... other detection logic
  }
}
```
